Alerts

Backbone.BUI.Alert

Error or danger

var alertError = new Backbone.BUI.Alert({
  ctype: Backbone.BUI.Config.Alert.ERROR,
  title: 'This is a test title',
  message: 'This is a test message',
  renderTo: this.$('#alert-error'),
  timeout: null
});
alertError.render();
